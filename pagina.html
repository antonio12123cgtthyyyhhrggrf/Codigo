23. Imprimir los 10 primeros números primos

Python

def es_primo(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True

contador = 0
num = 2
while contador < 10:
    if es_primo(num):
        print(num)
        contador += 1
    num += 1

JavaScript

function esPrimo(n) {
  if (n < 2) return false;
  for (let i = 2; i <= Math.sqrt(n); i++) {
    if (n % i === 0) return false;
  }
  return true;
}

let contador = 0;
let num = 2;
while (contador < 10) {
  if (esPrimo(num)) {
    console.log(num);
    contador++;
  }
  num++;
}
